<script>
function toggleMenu(el){
    const submenu = el.nextElementSibling;
    const arrow = el.querySelector(".arrow");
    if(submenu){
        submenu.classList.toggle("open");
        arrow.textContent = submenu.classList.contains("open") ? "▼" : "▶";
    }
}

function showHome(){
    document.getElementById('homeContent').style.display = 'block';
    document.getElementById('lessonContent').style.display = 'none';
}

function loadLesson(curriculum, grade, lesson){
    document.getElementById('homeContent').style.display = 'none';
    const lessonDiv = document.getElementById('lessonContent');
    lessonDiv.style.display = 'block';

    lessonDiv.innerHTML = `
        <div class="breadcrumbs">
            <span>Home</span>
            <span>${curriculum}</span>
            <span>${grade}</span>
            <span>${lesson}</span>
        </div>

        <div class="top-header1">
            <h1>${lesson}</h1>
        </div>

        <h2>${lesson} Content</h2>

        <br>

        <label>Your Progress:</label><br>
        <input type="range" min="0" max="100" step="10" value="0" id="rangeBar">
        <p id="percentText">0%</p>

        <div class="progress-container">
            <div class="progress-bar" id="progressBar"></div>
        </div>

        <br><br>

        <h2>
        <a href="#" onclick="loadSubject('${curriculum}','${grade}','${lesson}','Mathematics')">
        Mathematics - ${lesson}
        </a>
        </h2>

        <br>

        <button onclick="showHome()" class="B">Back to Home</button>
    `;

    attachProgress();
}

function loadSubject(curriculum, grade, lesson, subject){
    const lessonDiv = document.getElementById('lessonContent');

    lessonDiv.innerHTML = `
        <div class="breadcrumbs">
            <span>Home</span>
            <span>${curriculum}</span>
            <span>${grade}</span>
            <span>${lesson}</span>
            <span>${subject}-${lesson}</span>
        </div>

        <div class="top-header1">
            <h1>${subject} - ${lesson}</h1>
        </div>

        <h2>${subject} Detailed Content</h2>

        <br>

        <label>Your Progress:</label><br>
        <input type="range" min="0" max="100" step="10" value="0" id="rangeBar">
        <p id="percentText">0%</p>

        <div class="progress-container">
            <div class="progress-bar" id="progressBar"></div>
        </div>

        <br>

        <button onclick="showHome()" class="B">Back to Home</button>
    `;

    attachProgress();
}

function attachProgress(){
    const range = document.getElementById("rangeBar");
    const progressBar = document.getElementById("progressBar");
    const percentText = document.getElementById("percentText");

    range.addEventListener("input", function(){
        progressBar.style.width = this.value + "%";
        percentText.innerText = this.value + "%";
    });
}
